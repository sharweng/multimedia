About This Project

This solar system animation was created using Blender 3D with custom Python scripts to accurately simulate planetary motion. The project combines artistic visualization with scientific accuracy to create an educational and visually stunning representation of our solar system.

The Python script handles the orbital mechanics, calculating the position of each planet based on Kepler's laws of planetary motion. This ensures that the relative speeds and distances of the planets are accurate, though the scale has been adjusted to make the visualization more effective.

Each planet was textured using high-resolution NASA imagery, and the lighting was set up to simulate the actual illumination from the sun. Special attention was paid to details like the rings of Saturn, the Great Red Spot on Jupiter, and the atmospheric effects on each planet.

Python Script Highlights

import bpy
import math
from mathutils import Vector

# Clear existing objects
bpy.ops.object.select_all(action='DESELECT')
bpy.ops.object.select_by_type(type='MESH')
bpy.ops.object.delete()

# Create Sun
sun = bpy.ops.mesh.primitive_uv_sphere_add(radius=2, location=(0, 0, 0))
sun_obj = bpy.context.active_object
sun_obj.name = 'Sun'

# Create material for Sun
sun_mat = bpy.data.materials.new(name="SunMaterial")
sun_mat.use_nodes = True
nodes = sun_mat.node_tree.nodes
bsdf = nodes.get("Principled BSDF")
bsdf.inputs[0].default_value = (1.0, 0.8, 0.0, 1.0)  # Yellow color
bsdf.inputs[19].default_value = 5.0  # Emission strength
sun_obj.data.materials.append(sun_mat)

# Planet data: name, radius, distance from sun, orbit period (days)
planets = [
    ("Mercury", 0.2, 5, 88),
    ("Venus", 0.3, 7, 225),
    ("Earth", 0.4, 10, 365),
    ("Mars", 0.3, 13, 687),
    ("Jupiter", 0.8, 20, 4333),
    ("Saturn", 0.7, 25, 10759),
    ("Uranus", 0.5, 30, 30687),
    ("Neptune", 0.5, 35, 60190)
]

# Create planets
for name, radius, distance, period in planets:
    bpy.ops.mesh.primitive_uv_sphere_add(radius=radius, location=(distance, 0, 0))
    planet = bpy.context.active_object
    planet.name = name
    
    # Create orbit path
    bpy.ops.curve.primitive_bezier_circle_add(radius=distance)
    orbit = bpy.context.active_object
    orbit.name = f"{name}_Orbit"
    
    # Set up animation
    frames_per_orbit = period / 2  # Scaled for animation
    bpy.context.scene.frame_end = max(bpy.context.scene.frame_end, int(frames_per_orbit))
    
    # Animate planet along orbit
    for frame in range(int(frames_per_orbit)):
        angle = (frame / frames_per_orbit) * 2 * math.pi
        x = distance * math.cos(angle)
        y = distance * math.sin(angle)
        planet.location = (x, y, 0)
        planet.keyframe_insert(data_path="location", frame=frame)

# Set up camera and lighting
bpy.ops.object.camera_add(location=(0, -50, 30), rotation=(math.radians(60), 0, 0))
camera = bpy.context.active_object

# Add a track to constraint to follow the sun
track_to = camera.constraints.new('TRACK_TO')
track_to.target = sun_obj
track_to.track_axis = 'TRACK_NEGATIVE_Z'
track_to.up_axis = 'UP_Y'

# Set up rendering
bpy.context.scene.render.resolution_x = 3840
bpy.context.scene.render.resolution_y = 2160
bpy.context.scene.render.film_transparent = True

print("Solar system setup complete!")
